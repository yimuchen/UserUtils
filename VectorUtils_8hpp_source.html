64RemoveDuplicate( const std::vector<T>& dupvec )
65{
66 std::set<T> unique_set;
67 std::vector<T> ans;
68
69 for( const auto& item : dupvec ){

179ListFromFile( const std::string& file, const std::string& delimiter )
180{
181 std::ifstream f( file );
182 std::stringstream buffer;
183 std::string content;
184 std::vector<std::string> tokens;
185 std::vector<OBJTYPE> ans;
186
187 // Getting the full contents of the file
188 buffer << f.rdbuf();
189 content = buffer.str();
190
191 // Boost algorithm for string splitting
192 boost::split( tokens, content, boost::is_any_of( delimiter ) );
193
194 // std algorithm for removing empty strings.
195 tokens.erase( std::remove( tokens.begin(), tokens.end(), "" ), tokens.end() );
196
197 for( const auto& t : tokens ){
198 ans.push_back( boost::lexical_cast<OBJTYPE>( t ) );
199 }
200
201 return ans;
202}
