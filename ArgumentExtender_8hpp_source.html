210template<typename T>
211T
212ArgumentExtender::Arg( const std::string& opt ) const
213{
214 if( !CheckArg( opt ) ){
215 throw std::invalid_argument(
216 usr::fstr( "Option [%s] was not provided as a program options, "
217 ", see --help output", opt ) );
218 }
219 return _argmap[opt].as<T>();
220}
221
222template<typename T>
223T
224ArgumentExtender::ArgOpt( const std::string& opt, const T& val ) const
225{ return CheckArg( opt ) ? Arg<T>( opt ) : val; }
226
227template<typename T>
228std::vector<T>
229ArgumentExtender::ArgList( const std::string& opt ) const
230{
231 if( CheckArg( opt + "_list" ) ){
232 const std::string txtfile = Arg<std::string>( opt+"_list" );
233 return ListFromFile<T>( txtfile );
234 } else if( !CheckArg( opt ) ){
235 throw std::invalid_argument(
236 usr::fstr( "Option [%s] was not provided as a program options, "
237 ", see --help output", opt ) );
238 } else {
239 return _argmap[opt].as<std::vector<T> >();
240 }
241}
